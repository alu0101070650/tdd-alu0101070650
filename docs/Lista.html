def each
  nodo_actual = @cabeza
  1.upto(numero_nodos) do
    yield nodo_actual.valor
    nodo_actual = nodo_actual.siguiente
  end
end

def extraer_cabeza
  if @cabeza.nil?
    nil
  else
    antigua_cabeza = @cabeza
    siguiente = @cabeza.siguiente
    if siguiente.nil?
      @cabeza = nil
      @cola = nil
    else
      siguiente.previo = nil
      @cabeza = siguiente
    end
    antigua_cabeza.valor
  end
end

def insertar(elemento)
  nodo = Nodo.new
  nodo.valor = elemento
  if @cabeza.nil?
    @cabeza = nodo
    @cola = nodo
  else
    @cola.siguiente = nodo
    nodo.previo = @cola
    @cola = nodo
  end
end

def insertar_a(elementos)

def numero_nodos
  nodo_actual = @cabeza
  numero_nodos = 0
  while nodo_actual != nil do
    nodo_actual = nodo_actual.siguiente
    numero_nodos += 1
  end
  numero_nodos
end
